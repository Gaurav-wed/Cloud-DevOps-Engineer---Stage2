apiVersion: apps/v1
kind: Deployment
metadata:
  name: sample-website
  labels:
    app: sample-website
spec:
  replicas: 2
  selector:
    matchLabels:
      app: sample-website
  template:
    metadata:
      labels:
        app: sample-website
    spec:
      containers:
        - name: nginx
          image: nginx:latest
          ports:
            - containerPort: 80
          volumeMounts:
            - name: website-content
              mountPath: /usr/share/nginx/html
      volumes:
        - name: website-content
          configMap:
            name: sample-website-config
---
apiVersion: v1
kind: Service
metadata:
  name: sample-website-service
  labels:
    app: sample-website
spec:
  type: NodePort
  selector:
    app: sample-website
  ports:
    - protocol: TCP
      port: 80
      targetPort: 80
      nodePort: 30001
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: sample-website-config
data:
  index.html: |
    <!DOCTYPE html>
    <html>
    <head>
        <title>Sample Website</title>
    </head>
    <body>
        <h1>Welcome to the Sample Website!</h1>
        <p>This is a simple Nginx website running on Kubernetes.</p>
    </body>
    </html>
